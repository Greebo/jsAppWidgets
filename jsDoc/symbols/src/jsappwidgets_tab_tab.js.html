 24  * @example var myTab = new TabElement(   "myTab",           
 25           { cssStyle         : 'simple',  // css class of the div

390   create : function()
391   {
392     // new Object for Folders
393     if ( !this.m_oRegister )
394       this.m_oRegister = new TabRegister( this.m_nSource, this.m_hStyle, this );
395     // new Box
396     if ( !this.m_oBox )
397       this.m_oBox      = new TabBox(      this.m_nSource, this.m_hStyle, this );
398 
399     // a div over the whole tab
400     this.m_nTab = new Element( "table", {cellSpacing: 0, cellPadding : 0} );
401     this.m_nTab.setStyle( { left   : this.m_hStyle.get('left'),
402                             top    : this.m_hStyle.get('top')
403                           } );
404 
405 
406     //insert the two TabElements (register + box) in the node
407     v_nRegisterTD = new Element("td");
408     v_nRegisterTD.appendChild( this.m_oRegister.element );  
409     v_nBoxTD      = new Element( "td" );
410     v_nBoxTD.appendChild( this.m_oBox.element );
411 
412     // which order? first register than box? or reverse?
413     v_nTR = this.m_nTab.insertRow( 0 );
414     switch ( this.m_hStyle.get('regPos'))
415     {
416       case 'top' :
417         v_nTR.appendChild( v_nRegisterTD );
418         this.m_nTab.insertRow( -1 ).appendChild( v_nBoxTD );
419         break;
420       case 'bottom' :
421         v_nTR.appendChild( v_nBoxTD );
422         this.m_nTab.insertRow( -1 ).appendChild( v_nRegisterTD );
423         break;
424     }
425     // return the new created node
426     return this.m_nTab;
427   },

557   // subfunction from constructor
558   initStyle : function( $super, p_hStyle )
559   {
560     $super( p_hStyle );
561     
562     // limit register counts
563     if ( this.m_hStyle.get( 'regRows' ) > this.m_iCntRegPages )
564       this.m_hStyle.set( 'regRows', this.m_iCntRegPages);
565     if ( this.m_hStyle.get( 'regRows' ) > 100 ) 
566       this.m_hStyle.set( 'regRows', 100); 
567 
568     // Define register page colors
569     this.m_sRegColTbl = new Array( "", "white", "white", "gray", "black", this.m_hStyle.get('backgroundColor'), this.m_hStyle.get('backgroundColor'));
570     // change color if bottom register
571     if( this.m_hStyle.get('regPos') == "bottom" )
572     {
573       this.m_sRegColTbl[2] = "black";
574       this.m_sRegColTbl[6] = "gray";
575     }
576 
577     // color table for register pages (active & non-active)(top+bottom)
578     this.m_aRegPageColTbl    = new Array();
579     this.m_aRegPageColTbl[0] = new Array( 0, 0, 2, 2, 0, 0 );
580     this.m_aRegPageColTbl[1] = new Array( 0, 1, 6, 6, 4, 0 );
581     this.m_aRegPageColTbl[2] = new Array( 1, 5, 5, 5, 3, 4 );
582     //    These are set at creation time (look further in code...)
583     //this.m_aRegPageColTbl[3] = new Array( 2, 2, 2, 2, 2, 2 ); <- non-active
584     //this.m_aRegPageColTbl[3] = this.m_aRegPageColTbl[2];      <- active
585     
586     // Position of GFX and NAME in RegPageTbl
587     this.m_aGfxPos  = ( { x: 2, y : 2 } );
588     this.m_aNamePos = ( { x: 3, y : 2 } );

590   /**
591    * @function
592    */
593    create : function()
594    {
595      // table over the register rows
596      var v_nTable = new Element( "table", { cellSpacing : 0, cellPadding : 0 } );
597      v_nTable.setStyle( { width : '100%' } );
598 
599      var v_nTR = "";
600      var v_nTD = "";
601      var v_iCntRows = this.m_hStyle.get( 'regRows' );
602      var v_fOffset = ( this.m_iCntRegPages / v_iCntRows );
603      
604      // build all register rows in a table
605      for ( var v_iRow = 1; v_iRow <= v_iCntRows; v_iRow++ )
606      {
607        // create new register row
608        v_iStart = ~~( v_fOffset * (v_iRow-1)) + 1; // start page of register
609        v_iEnd   = ~~( v_fOffset * v_iRow);         // end page of register
610        v_bActPageInRegister = ( v_iStart <= this.m_iActPage && v_iEnd >= this.m_iActPage ); // flag (is active page in this register row?)
611        v_nRegisterRow = this.createRow( v_iStart, v_iEnd, v_bActPageInRegister );
612 
613        if ( !v_bActPageInRegister )
614        {
615          // make overlapping ( go <n> pixels down )
616          v_sDistance = ~~(this.m_hStyle.get('regDistance'));
617          if ( this.m_hStyle.get('regPos')  == 'bottom' ) 
618            v_sDistance *= (-1);
619          
620          v_nRegisterRow.setStyle( { top      : v_sDistance +'px', 
621                                     position :'relative', 
622                                     zIndex   : this.m_hStyle.get('regRows') - v_iRow } );
623 
624          v_nTR = v_nTable.insertRow( (-1) * (this.m_hStyle.get('regPos') == 'bottom') );
625        }
626        else // first line (front line)
627        {
628          v_nRegisterRow.setStyle( { position : 'relative', 
629                                     zIndex   : this.m_hStyle.get( 'regRows' ) } );
630          
631          v_nTR = v_nTable.insertRow( (-1) * (this.m_hStyle.get('regPos') == 'top'  ) );
632        }
633 
634        // append in table
635        v_nTD = new Element( "td" );
636       
637        v_nTD.appendChild( v_nRegisterRow );
638           
639        v_nTR.appendChild( v_nTD );

646    createRow : function( p_iLeft, p_iRight, p_bActPageInRegister )
647    {
648      // table over all page Registers
649      var v_nTable = new Element( "table", { cellSpacing : 0, cellPadding : 0 } );
650      v_nTable.setStyle( { width : '100%' } );
651 
652      // new row in table
653      var v_nTR = $(v_nTable.insertRow(0));
654      var v_sRegAlign = this.m_hStyle.get('regAlign');
655      
656      // left filler    
657      if ( this.m_iActPage != p_iLeft || v_sRegAlign == 'right' )
658          v_nTR.appendChild( this.createFiller( true, p_bActPageInRegister ) ); // true == left filler
659 
660      // draw all pages
661      for( i = p_iLeft; i <= p_iRight; i++ )
662      {
663        // td for page
664        var v_nTD = new Element( "td" );
665        if( this.m_hStyle.get('regPos') == 'bottom' )
666          v_nTD.vAlign = "top";
667        else
668          v_nTD.vAlign = "bottom";
669        
670        // make the register page        
671        v_nRegister = this.createPage( i, p_bActPageInRegister );
672 
673        // anker the register page in td of Register
674        v_nTD.appendChild( v_nRegister );
675        
676        // anker td in row of Register
677        v_nTR.appendChild( v_nTD );
678      }
679      
680      // right filler (createFiller returns the td node)
681      if ( this.m_iActPage != p_iRight || v_sRegAlign == 'left' ) 
682        v_nTR.appendChild( this.createFiller( false, p_bActPageInRegister ) );
683 
684      // return the new Register node
685      return v_nTable;
686    },

689     * create the left or ritht filler. 
690     * 
691     * @param {boolean} create left filler ( false : create right filler)
692     * 
693     * @return filler td node 
694     * @private
695     */
696    createFiller : function( p_bLeft, p_bDrawBorder )
697    {
698      // new td node for the filler
699      v_nTD = new Element("td").addClassName( "filler" );
700        
701      // stretch  filler?
702      if(  (  p_bLeft && this.m_hStyle.get( 'regAlign' ) == 'right' )
703        || ( !p_bLeft && this.m_hStyle.get( 'regAlign' ) == 'left'  ) )
704        v_nTD.setStyle( { width : "100%" } );
705 
706      // bottom or top border
707      if ( p_bDrawBorder ) // only in front row of register the border must be visible
708      {
709        switch( this.m_hStyle.get('regPos') )
710        {
711          case 'bottom': v_nTD.setStyle( { borderTop    : "1px solid black" } );  break;
712          case 'top'   : v_nTD.setStyle( { borderBottom : "1px solid white" } ); break;
713        }
714      }
715 
716      // for compatibility ie 6
717      v_nBox = new Element( "div" ).setStyle( {width : "1px"} );
718      v_nTD.appendChild( v_nBox );
719   
720      // return the new filler
721      return v_nTD;

724  /**
725   * draw a page in the register
726   * 
727   * @param i number of the tab
728   * 
729   * @returns the node
730   * 
731   * @private 
732   */
733   createPage : function( p_iPage, p_bCutBorder )
734   {
735     var v_sSelected = "";
736     // Page is the actual one?
737     if ( p_iPage == this.m_iActPage )
738     {
739       v_sSelected = "Selected"; 
740       this.m_aRegPageColTbl[3] = this.m_aRegPageColTbl[2]; // no border line
741     }
742     else // not actual
743       this.m_aRegPageColTbl[3] = new Array( 2, 2, 2, 2, 2, 2 ); // a white or black line
744  
745     // Set the ColorColor:
746     this.m_sRegColTbl[5] = this.getColor( p_iPage );   
747       
748     // generate the table element for the page
749     var v_nTable = new Element( "table", { cellSpacing: 0, cellPadding : 0 });
750     v_nTable.setStyle({ width : "100%" });  // needed for chrome
751 
752     // start and end point of register page (cut the border if page is neighbor of active page
753     // and the active page is in same row
754     v_iStartColmn = ( p_iPage == this.m_iActPage + 1 && p_bCutBorder ? 2 : 0 );
755     v_iEndColmn   = ( p_iPage == this.m_iActPage - 1 && p_bCutBorder ? 3 : 5 );
756 
757     // 4 rows (only 3 if the row is not the first in the front)
758     for( var v_iRow = 0 ; v_iRow < (p_bCutBorder ? 4 : 3); v_iRow++ )
759     {
760       // new TR on first child or last
761       v_nTR = $(v_nTable.insertRow( (this.m_hStyle.get('regPos') == 'top') * v_iRow ));
762         
763       for( var v_iClmn = v_iStartColmn; v_iClmn <= v_iEndColmn; v_iClmn++ )
764       {
765         // new td in tr with correct background color
766         var v_nTD  = new Element( "td" ).setStyle( {
767                      backgroundColor : this.m_sRegColTbl[ this.m_aRegPageColTbl[v_iRow][v_iClmn] ],
768                      paddingLeft     : "1px",
769                      paddingTop      : "1px"  } );
770         v_nTR.appendChild( v_nTD );
771 
772         // gfx node must be filled
773         if ( v_iRow == this.m_aGfxPos.y && v_iClmn == this.m_aGfxPos.x && this.m_aRegGfx[p_iPage] ) 
774         {
775           v_nTD.addClassName( "tabsGfx" + v_sSelected );
776           v_nTD.appendChild( new Element( "img", { 'src' : this.m_aRegGfx[p_iPage] +''}  ).setStyle( { border : 'none' } ) );
777         }
778       //Text node
779         else if( v_iRow == this.m_aNamePos.y && v_iClmn == this.m_aNamePos.x ) 
780         {
781           v_nTD.addClassName( "tabsName" + v_sSelected );
782           v_nTD.setStyle( { width : "100%" } );
783           v_nTD.appendChild( document.createTextNode( this.m_aRegName[p_iPage] ));
784           v_nTable.style.whiteSpace = 'nowrap';
785         }
786       } //for : all columns
787     } // for : all Rows
788 
789     // <a> tag over the table    
790     v_nA = new Element( "a" );
791     v_nA.className = this.m_aRegActive[p_iPage] == 1  ? "tabsName" + v_sSelected : "tabsNameInactive";
792     if ( p_iPage == this.m_iActPage )
793       v_nA.href = "javascript:void(0)";
794     else
795       v_nA.href = "javascript:setPage( " + this.m_iTabNr + ", " + p_iPage + " );";
796     // for ie : need to activate the <a> tag!
797     v_nTable.observe('click', linkClick );
798     v_nA.appendChild( v_nTable );
799 
800     // return the new register node
801     return v_nA;
802   },

807     // the registers must be refreshed    
808     this.redraw();
